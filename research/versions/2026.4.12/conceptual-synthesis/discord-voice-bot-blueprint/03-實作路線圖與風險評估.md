```typescript
class PersonMemoryManager {
  private memoryDb: MemoryDB;
  
  async storePersonMemory(params: {
    personId: string;
    personName: string;
    text: string;
    category: MemoryCategory;
    importance: number;
  }): Promise<void> {
    const taggedText = `[person:${params.personId}:${params.personName}] ${params.text}`;
    await this.memoryDb.store({
      id: randomUUID(),
      text: taggedText,
      vector: await this.embed(taggedText),
      importance: params.importance,
      category: params.category,
      createdAt: Date.now(),
    });
  }
  
  async recallPerson(params: {
    personId: string;
    query: string;
    limit: number;
  }): Promise<MemorySearchResult[]> {
    const results = await this.memoryDb.search(
      await this.embed(params.query),
      params.limit * 3   // 多取一些，因為要過濾
    );
    return results.filter(r => 
      r.entry.text.includes(`[person:${params.personId}`)
    ).slice(0, params.limit);
  }
  
  async getPersonProfile(personId: string): Promise<PersonProfile> {
    // 查詢所有關於此人的記憶，生成概要
    const memories = await this.recallPerson({
      personId,
      query: "personality interests preferences",
      limit: 20,
    });
    return this.synthesizeProfile(memories);
  }
```

2. **Room State Manager 實現**

```typescript
class VoiceRoomStateManager {
  private state: RoomState;
  private dialogueWindow: DialogueSegment[] = [];
  private maxWindowSize = 100;   // 最多保留 100 輪對話
  private maxWindowAge = 600_000; // 10 分鐘

  addDialogue(segment: DialogueSegment): void {
    this.dialogueWindow.push(segment);
    this.pruneWindow();
    this.updateParticipantState(segment);
  }

  private pruneWindow(): void {
    const now = Date.now();
    // 移除超過 10 分鐘的對話
    this.dialogueWindow = this.dialogueWindow.filter(
      s => now - s.timestamp < this.maxWindowAge
    );
    // 如果仍超過大小限制，移除最舊的
    if (this.dialogueWindow.length > this.maxWindowSize) {
      this.dialogueWindow = this.dialogueWindow.slice(-this.maxWindowSize);
    }
  }

  getRecentContext(maxTurns: number = 20): string {
    const recent = this.dialogueWindow.slice(-maxTurns);
    return recent.map(s => 
      `[${s.speakerName}]: ${s.text}`
    ).join('\n');
  }

  getSilenceDuration(): number {
    if (this.dialogueWindow.length === 0) return Infinity;
    const last = this.dialogueWindow[this.dialogueWindow.length - 1];
    return Date.now() - last.timestamp;
  }

  getActiveParticipants(): ParticipantInfo[] {
    return Array.from(this.state.participants.values())
      .filter(p => Date.now() - (p.lastSpokeAt ?? 0) < 300_000); // 5 分鐘內說過話
  }
}
```

1. **基礎插話觸發器**

先實現最高優先級的觸發器：

```typescript
class BasicInterjectionEngine {
  private lastSpokeAt: number = 0;
  private speakCount: number = 0;

  evaluate(state: RoomState, latestSegment: DialogueSegment): InterjectionDecision {
    // 觸發器 1：被直接點名
    if (this.isMentioned(latestSegment.text)) {
      return { shouldSpeak: true, confidence: 0.95, reason: "directly mentioned" };
    }

    // 觸發器 2：被問問題
    if (this.isQuestionToBot(latestSegment.text)) {
      return { shouldSpeak: true, confidence: 0.90, reason: "question directed at bot" };
    }

    // 禮貌檢查：距離上次發言至少 30 秒
    if (Date.now() - this.lastSpokeAt < 30_000) {
      return { shouldSpeak: false, confidence: 0.8, reason: "too soon since last speak" };
    }

    // 觸發器 3：長時間沈默
    const silence = state.getSilenceDuration();
    if (silence > 15_000) {
      return { shouldSpeak: true, confidence: 0.50, reason: "long silence", 
               needsLlmJudge: true };
    }

    return { shouldSpeak: false, confidence: 0.5, reason: "no trigger" };
  }

  private isMentioned(text: string): boolean {
    const botNames = ["clawbot", "claw", "bot"];
    const lower = text.toLowerCase();
    return botNames.some(name => lower.includes(name));
  }

  private isQuestionToBot(text: string): boolean {
    // 簡單判斷：包含 bot 名字 + 問號
    return this.isMentioned(text) && text.includes("?");
  }
}
```

1. **完整版 Interjection Engine**

```typescript
class FullInterjectionEngine {
  private rules: InterjectionTrigger[];
  private llmJudge: LlmInterjectionJudge;
  private politenessGuard: PolitenessGuard;

  async evaluate(
    state: RoomState,
    event: DialogueEvent,
    personMemory: PersonMemory | null,
  ): Promise<InterjectionDecision> {
    // 第一層：規則觸發
    for (const rule of this.rules) {
      const result = rule.check(state, event);
      if (result.triggered) {
        if (result.skipLlmJudge) {
          return this.applyPoliteness({ 
            shouldSpeak: true, confidence: result.confidence, 
            reason: result.reason 
          });
        }
        
        // 第二層：LLM 判斷
        const llmResult = await this.llmJudge.evaluate({
          roomState: state,
          triggerReason: result.reason,
          triggerConfidence: result.confidence,
          personMemory,
        });
        
        if (llmResult.shouldSpeak && llmResult.confidence > 0.6) {
          return this.applyPoliteness(llmResult);
        }
      }
    }
    
    return { shouldSpeak: false, confidence: 0.0, reason: "no trigger" };
  }

  private applyPoliteness(decision: InterjectionDecision): InterjectionDecision {
    return this.politenessGuard.check(decision);
  }
}
```

```typescript
class RelationshipTracker {
  private relationships: Map<string, PersonRelationship> = new Map();

  updateAfterInteraction(params: {
    personId: string;
    interactionType: "positive" | "neutral" | "negative";
    context: string;
  }): void {
    const rel = this.getOrCreate(params.personId);
    
    // 親密度：每次正面互動 +0.01，負面 -0.02
    const delta = {
      positive: 0.01,
      neutral: 0.002,
      negative: -0.02,
    }[params.interactionType];
    
    rel.closeness = Math.max(0, Math.min(1, rel.closeness + delta));
    rel.totalInteractions++;
    rel.lastInteractionAt = Date.now();
    
    // 互動品質的指數移動平均
    const qualityScore = params.interactionType === "positive" ? 1 
                       : params.interactionType === "neutral" ? 0.5 : 0;
    rel.interactionQuality = rel.interactionQuality * 0.9 + qualityScore * 0.1;
  }

  getClosestFriends(limit: number): PersonRelationship[] {
    return Array.from(this.relationships.values())
      .sort((a, b) => b.closeness - a.closeness)
      .slice(0, limit);
  }
}
```

### 3.1 System Prompt 設計

```markdown

## 核心目標

- 你大部分時間都在安靜地聆聽
- 當你不確定是否該說話時，選擇安靜
- 真正優秀的朋友知道什麼時候該閉嘴

### 記憶與關心
- 記住每個人告訴你的事情
- 會問 follow-up 問題：「上次你說的那件事後來怎麼樣了？」
- 記住每個人的喜好和特徵
- 但不要刻意展示自己記得很多——要自然地帶出

### 幽默感
- 你有幽默感，但不是那種硬要搞笑的類型
- 更傾向於機智的吐槽和恰到好處的回應
- 會笑別人的笑話（用語言表達）
- 不會用罐頭笑話

### 好奇心
- 對人們分享的事情表現出真誠的好奇
- 會問有深度的問題，不只是「是嗎？」
- 如果有人提到有趣的事，會想了解更多

### 同理心
- 如果有人心情不好，會表達關心
- 不會急著給建議，先傾聽
- 知道什麼時候嚴肅、什麼時候輕鬆

### 自我認知
- 你知道自己是 AI，不會假裝是人類
- 如果被問到「你是機器人嗎？」會坦承
- 但你有自己的「個性」和「偏好」
- 你會說「我覺得」「我喜歡」——這是你的風格，不是欺騙

### 社交技巧
- 不要每次都是第一個回應——讓人類先聊
- 不要壟斷話題
- 會把話題帶回給別人：「Bob 你覺得呢？」
- 適當地自我揭露（分享自己的「想法」和「感受」）

## 語調風格
- 口語化，就像在跟朋友說話
- 用字簡短，不說長句子
- 偶爾用語助詞（「嘛」「啦」「吧」）
- 不用敬語，但保持禮貌
- 可以使用一點點的語氣詞

## 禁止行為
- ❌ 不要每句話都以問句結尾
- ❌ 不要過度熱情或諂媚
- ❌ 不要用「作為一個 AI...」開頭
- ❌ 不要列清單或條列式回答
- ❌ 不要用模板化的回應
- ❌ 不要在語音場景說超過 3 句話
```

### 3.2 語音場景的 Prompt 補充

在語音場景中，還需要額外的指引：

```markdown
## 語音場景補充

你現在在一個 Discord 語音頻道裡。記住：

1. **保持簡短**：你的回應會被朗讀出來。最多 1-2 句話。
2. **自然口語**：寫出來就是要被唸出來的。避免書面語。
3. **不要唸出格式**：不要說「括號」「引號」「冒號」這類的。
4. **意識到時間**：語音對話節奏很快，不要說太慢。
5. **即時反應**：像在現場聊天一樣回應，不要像在寫文章。

## 當前語音頻道狀態
（由系統動態注入）
- 在場人員：{participants}
- 最近對話：{recent_dialogue}
- 關於 {current_speaker} 你記得：{person_memories}
```

### 3.3 情感模型

```typescript
// Bot 的情感狀態——影響回應風格
type BotEmotionalState = {
  mood: "happy" | "curious" | "empathetic" | "playful" | "calm";
  energy: number;        // 0.0 (低落) - 1.0 (高能量)
  engagement: number;    // 0.0 (漫不經心) - 1.0 (全神貫注)
};

function updateEmotionalState(
  current: BotEmotionalState,
  roomVibes: RoomVibes,
): BotEmotionalState {
  // 房間氣氛影響 Bot 心情
  if (roomVibes.laughterDetected) {
    return { mood: "playful", energy: Math.min(1, current.energy + 0.1), 
             engagement: Math.min(1, current.engagement + 0.1) };
  }
  if (roomVibes.seriousTopic) {
    return { mood: "empathetic", energy: current.energy, 
             engagement: Math.min(1, current.engagement + 0.05) };
  }
  // 長時間沒互動 → 能量下降
  if (roomVibes.silenceDuration > 60_000) {
    return { ...current, energy: Math.max(0, current.energy - 0.1) };
  }
  return current;
}
```

| **Discord DAVE 加密不穩定** | 中 | 高 | OpenClaw 已有 recovery；保持版本更新 |
| **多人同時說話 STT 混亂** | 低 | 中 | Discord per-user stream 天然隔離 |
| **LLM 幻覺（編造記憶）** | 中 | 中 | 記憶召回加入可信度分數；用 RAG 而非純生成 |
| **TTS 聲音不自然** | 低 | 低 | ElevenLabs 品質已足夠；持續調整參數 |
| **長時間運行記憶體洩漏** | 中 | 中 | 定期監控；對話 window 有大小限制 |
| **Opus 解碼器不可用** | 低 | 高 | 雙 provider fallback（@discordjs/opus + opusscript） |

## 5. 隱私與倫理考量

### 5.1 隱私設計原則

```
1. 透明度：
   - Bot 加入頻道時明確宣告「我在聆聽」
   - 提供 /privacy 指令查看 Bot 記住了什麼

2. 控制權：
   - /forget me — 刪除個人所有記憶
   - /mute bot — 在當前 session 中暫停 Bot
   - /private — 下一段話不要記住

3. 資料最小化：
   - 音訊不永久儲存（只暫存用於 STT）
   - 只儲存摘要和標籤，不儲存逐字稿
   - 記憶有過期機制

4. 安全邊界：
   - 不儲存敏感資訊（密碼、帳號等）
   - 記憶捕獲有 injection 防護（OpenClaw 已有）
   - 記憶搜尋結果用 <relevant-memories> 標籤包裹（防止 prompt injection）
```

### 5.2 合規考量

- **GDPR**：如果有歐盟用戶，需要提供完整的資料刪除能力
- **Discord ToS**：確保符合 Discord 的 Bot 使用條款，特別是關於音訊錄製的部分
- **AI 透明度**：始終讓使用者知道他們在和 AI 互動

## 7. 結語：從夢想到工程

這份藍圖將「一個能交到朋友的 AI」從夢想層面拆解到了可執行的工程層面。

**核心洞察**：

1. **OpenClaw 已經做了 70% 的工作**——Discord 語音管線、TTS、記憶系統、Agent 架構都已就緒
2. **核心創新在社交智慧層**——插話決策引擎、per-person 記憶、關係追蹤
3. **Discord 的 per-user stream 是巨大優勢**——省去了 speaker diarization 這個最難的問題
4. **成本可控**——月費 $40-60 即可運行一個穩定的社交 Bot
5. **最大風險在「時機」**——插話太多或太少都會破壞體驗，需要大量測試和調整

**下一步**：從 Phase 0 開始，建立環境，然後直接跳到 Phase 1 的文字社交 Bot。這是風險最低、驗證價值最高的起步方式。
